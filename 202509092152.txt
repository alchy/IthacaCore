/**
 * @brief Generuje envelope data pro danÃ½ sample rate
 */
void Envelope::generateEnvelopeForSampleRate(int sampleRate, bool isAttack, 
                                           std::vector<float>& output, int& numSamples) const {
    float time_ms = isAttack ? attack_time_ms_ : release_time_ms_;
    float time_seconds = time_ms / 1000.0f;
    
    // PoÄet samples pro danÃ½ Äas
    numSamples = static_cast<int>(sampleRate * time_seconds);
    numSamples = std::max(1, numSamples);  // Min 1 sample
    
    output.resize(numSamples);
    
    // ExponenciÃ¡lnÃ­ kÅ™ivka s tau = time_seconds / 3 (cca 95% za danÃ½ Äas)
    float tau = time_seconds / 3.0f;
    
    for (int i = 0; i < numSamples; ++i) {
        float t = (static_cast<float>(i) / static_cast<float>(numSamples - 1)) * time_seconds;
        
        if (isAttack) {
            // Attack: 0 â†’ 1 exponenciÃ¡lnÄ›
            output[i] = 1.0f - std::exp(-t / tau);
        } else {
            // Release: 1 â†’ 0 exponenciÃ¡lnÄ›
            output[i] = std::exp(-t / tau);
        }
        
        // Clamp na [0, 1]
        output[i] = std::max(0.0f, std::min(1.0f, output[i]));
    }
    
    // Truncate na threshold
    int converge_idx = -1;
    for (int i = 0; i < numSamples; ++i) {
        if (isAttack) {
            if (output[i] >= (1.0f - THRESHOLD)) {  // >= 0.99
                converge_idx = i;
                break;
            }
        } else {
            if (output[i] <= THRESHOLD) {  // <= 0.01
                converge_idx = i;
                break;
            }
        }
    }
    
    if (converge_idx > 0) {
        numSamples = converge_idx + 1;
        output.resize(numSamples);
    }
}

/**
 * @brief Helper: VyplnÃ­ buffer s envelope daty
 */
void Envelope::fillBufferFromFixedData(const FixedEnvelopeData& env, float* gainBuffer, 
                                      int numSamples, sf_count_t position, bool isAttack) const noexcept {
    // Vyber sprÃ¡vnÃ¡ data podle sample rate
    const float* src_data;
    int len;
    
    if (sample_rate_index_ == 0) {
        src_data = env.data_44100.get();
        len = env.len_44100;
    } else {
        src_data = env.data_48000.get();
        len = env.len_48000;
    }
    
    if (len <= 1) {
        // SpeciÃ¡lnÃ­ pÅ™Ã­pad pro velmi krÃ¡tkÃ© envelope
        float constant_value = isAttack ? 1.0f : 0.0f;
        std::fill(gainBuffer, gainBuffer + numSamples, constant_value);
        return;
    }
    
    // RT-safe vyplnÄ›nÃ­ s clampem
    for (int i = 0; i < numSamples; ++i) {
        sf_count_t idx = position + static_cast<sf_count_t>(i);
        
        // Clamp idx na [0, len-1]
        if (idx >= len) {
            // Po konci envelope: konstantnÃ­ target hodnota
            gainBuffer[i] = isAttack ? 1.0f : 0.0f;
        } else {
            gainBuffer[i] = src_data[static_cast<int>(idx)];
        }
    }
}

        }

    }

    // Pro debugging - toto nenÃ­ RT-safe, ale pro diagnostiku OK
    if (currentMismatch || loaderMismatch) {
        // Pokud je potÅ™eba reinicializace, debug info se vypÃ­Å¡e v changeSampleRate()
        return true;
    }
    
    return false;
}

/**

    if (needsReinitialization(targetSampleRate)) {
        changeSampleRate(targetSampleRate, logger);
    }
}

void VoiceManager::addActiveVoice(Voice* voice) noexcept {
    if (!voice) return;
    
    auto it = std::find(activeVoices_.begin(), activeVoices_.end(), voice);
    if (it == activeVoices_.end()) {
        activeVoices_.push_back(voice);
        activeVoicesCount_.fetch_add(1);
    }
}

void VoiceManager::removeActiveVoice(Voice* voice) noexcept {
    if (!voice) return;
    
    auto it = std::find(activeVoices_.begin(), activeVoices_.end(), voice);
    if (it != activeVoices_.end()) {
        activeVoices_.erase(it);
        activeVoicesCount_.fetch_sub(1);
    }
}

void VoiceManager::cleanupInactiveVoices() noexcept {
    for (Voice* voice : voicesToRemove_) {
        removeActiveVoice(voice);
    }
    
    voicesToRemove_.clear();
}

// convertFloatToInt16: Å kÃ¡luje a clipuje (bez ditheringu)
// SKonverze float â†’ int16 s manuÃ¡lnÃ­m clippingem (pro kompatibilitu s MSVC)
void WavExporter::convertFloatToInt16(float* src, int16_t* dst, int numSamples) {
    static constexpr int16_t MAX_INT16 = 32767;
    static constexpr int16_t MIN_INT16 = -32768;
    for (int i = 0; i < numSamples; ++i) {
        float scaled = src[i] * MAX_INT16;  // Å kÃ¡lovÃ¡nÃ­ -1.0..1.0 â†’ -32767..32767
        int32_t temp = static_cast<int32_t>(scaled);  // DoÄasnÃ½ int32_t pro bezpeÄnÃ½ cast
        int16_t clamped;
        if (temp > MAX_INT16) {
            clamped = MAX_INT16;  // Clipping nahoru
        } else if (temp < MIN_INT16) {
            clamped = MIN_INT16;  // Clipping dolÅ¯
        } else {
            clamped = static_cast<int16_t>(temp);  // Bez clippingu
        }
        dst[i] = clamped;
    }
}
