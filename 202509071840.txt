/**

    // ZÃ­skÃ¡nÃ­ metadat pÅ™es SamplerIO gettery

    // Krok 3: NaÄtenÃ­ dat pomocÃ­ sf_readf_float (automatickÃ¡ PCM->float konverze)

    } else {
        logger_.log("InstrumentLoader/loadSampleToBuffer", "info", 
                   "File already in 32-bit float format, no conversion needed: " + std::string(filename));
    }
    
    // Krok 4: NOVÃ‰ - Alokace permanent bufferu VÅ½DY pro stereo (frameCount * 2 * sizeof(float))
    size_t stereoBufferSize = frameCount * 2 * sizeof(float);
    float* permanentBuffer = static_cast<float*>(malloc(stereoBufferSize));
    
    if (!permanentBuffer) {
        logger_.log("InstrumentLoader/loadSampleToBuffer", "error", 
                   "Memory allocation error for permanent stereo buffer: " +

    if (channelCount == 1) {

        for (sf_count_t frame = 0; frame < frameCount; frame++) {
            permanentBuffer[frame * 2] = tempBuffer[frame];     // L kanÃ¡l
            permanentBuffer[frame * 2 + 1] = tempBuffer[frame]; // R kanÃ¡l (duplikace)
        }
        logger_.log("InstrumentLoader/loadSampleToBuffer", "info", 
                   "Mono to stereo conversion performed (L=R duplication): " + std::string(filename));
        
    } else if (channelCount == 2) {
        // STEREO â†’ STEREO: pÅ™Ã­mÃ© kopÃ­rovÃ¡nÃ­ (jiÅ¾ sprÃ¡vnÃ½ formÃ¡t)

            for (sf_count_t frame = 0; frame < frameCount; frame++) {
                permanentBuffer[frame * 2] = tempBuffer[frame];                    // L kanÃ¡l
                permanentBuffer[frame * 2 + 1] = tempBuffer[frameCount + frame];   // R kanÃ¡l
            }

// SKonverze float â†’ int16 s manuÃ¡lnÃ­m clippingem (pro kompatibilitu s MSVC)
void WavExporter::convertFloatToInt16(float* src, int16_t* dst, int numSamples) {
    static constexpr int16_t MAX_INT16 = 32767;
    static constexpr int16_t MIN_INT16 = -32768;
    for (int i = 0; i < numSamples; ++i) {
        float scaled = src[i] * MAX_INT16;  // Å kÃ¡lovÃ¡nÃ­ -1.0..1.0 â†’ -32767..32767
        int32_t temp = static_cast<int32_t>(scaled);  // DoÄasnÃ½ int32_t pro bezpeÄnÃ½ cast
        int16_t clamped;
        if (temp > MAX_INT16) {
            clamped = MAX_INT16;  // Clipping nahoru
        } else if (temp < MIN_INT16) {
            clamped = MIN_INT16;  // Clipping dolÅ¯
        } else {
            clamped = static_cast<int16_t>(temp);  // Bez clippingu
        }
        dst[i] = clamped;
    }
}
